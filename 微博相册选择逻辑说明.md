# 微博相册选择逻辑说明

## 一、UI结构分析

根据你提供的图片和描述，微博相册选择的UI结构如下：

```
androidx.recyclerview.widget.RecyclerView (根节点)
  ├─ FrameLayout[0] (第1张图片容器)
  │   ├─ ImageView (图片缩略图)
  │   ├─ FrameLayout (其他子元素)
  │   └─ TextView (作为checkbox，用于选中图片)
  ├─ FrameLayout[1] (第2张图片容器) ⭐ 目标
  │   ├─ ImageView
  │   └─ TextView (checkbox)
  ├─ FrameLayout[2] (第3张图片容器) ⭐ 目标
  │   ├─ ImageView
  │   └─ TextView (checkbox)
  └─ FrameLayout[3] (第4张图片容器)
      └─ ...
```

## 二、核心选择逻辑

### 关键步骤：

1. **查找RecyclerView根节点**
   ```kotlin
   AssistsCore.findByTags("androidx.recyclerview.widget.RecyclerView")
   ```

2. **遍历获取FrameLayout子项**
   ```kotlin
   for (index in 0 until recyclerView.childCount) {
       val child = recyclerView.getChild(index)
       if (TextUtils.equals("android.widget.FrameLayout", child.className)) {
           // 这是图片容器
       }
   }
   ```

3. **在FrameLayout中查找并点击TextView（checkbox）**
   ```kotlin
   frameLayout.findByTags("android.widget.TextView").forEach { textView ->
       // 点击TextView来选中图片
       textView.click() 或 textView.findFirstParentClickable()?.click()
   }
   ```

4. **选中第二张和第三张图片（索引1和2）**
   ```kotlin
   val targetIndices = listOf(1, 2)  // 第二张和第三张
   ```

## 三、完整实现代码

### 方式1: 直接在RecyclerView中遍历（推荐）

```kotlin
/**
 * 微博相册选择逻辑：选中第二张和第三张图片
 * UI结构: RecyclerView -> FrameLayout -> TextView(checkbox)
 */
private suspend fun selectWeiboImages(): Boolean {
    overlayLogging("开始查找微博相册图片，准备选中第二张和第三张")
    var selectedCount = 0
    val targetIndices = listOf(1, 2) // 第二张（索引1）和第三张（索引2）
    
    // 1. 查找RecyclerView根节点
    val recyclerViews = mutableListOf<AccessibilityNodeInfo>()
    recyclerViews.addAll(AssistsCore.findByTags("android.support.v7.widget.RecyclerView"))
    recyclerViews.addAll(AssistsCore.findByTags("androidx.recyclerview.widget.RecyclerView"))
    
    if (recyclerViews.isEmpty()) {
        overlayLogging("未找到RecyclerView")
        return false
    }
    
    // 2. 遍历每个RecyclerView
    for (recyclerView in recyclerViews) {
        overlayLogging("找到RecyclerView，开始遍历FrameLayout子项")
        val allFrameLayouts = mutableListOf<AccessibilityNodeInfo>()
        
        // 3. 收集所有FrameLayout子项
        for (index in 0 until recyclerView.childCount) {
            val child = recyclerView.getChild(index) ?: continue
            
            // 检查是否为FrameLayout
            if (TextUtils.equals("android.widget.FrameLayout", child.className)) {
                allFrameLayouts.add(child)
            }
        }
        
        overlayLogging("共找到${allFrameLayouts.size}个FrameLayout，准备选中第2张和第3张")
        
        // 4. 选中指定索引的图片（第二张和第三张）
        for (targetIndex in targetIndices) {
            if (targetIndex < allFrameLayouts.size) {
                val frameLayout = allFrameLayouts[targetIndex]
                val displayIndex = targetIndex + 1 // 显示用的索引（从1开始）
                
                overlayLogging("尝试选中第${displayIndex}张图片")
                val bounds = frameLayout.getBoundsInScreen()
                showClickEffect(bounds.centerX().toFloat(), bounds.centerY().toFloat(), "选中图片${displayIndex}")
                delay(200)
                
                // 5. 在FrameLayout中查找TextView（checkbox）并点击
                val selected = trySelectWeiboImage(frameLayout)
                if (selected) {
                    selectedCount++
                    overlayLogging("成功选中第${displayIndex}张图片")
                } else {
                    overlayLogging("未能选中第${displayIndex}张图片")
                }
                delay(300)
            } else {
                overlayLogging("图片索引${targetIndex + 1}不存在（总共只有${allFrameLayouts.size}张）")
            }
        }
        
        // 如果找到FrameLayout就退出循环
        if (allFrameLayouts.isNotEmpty()) break
    }
    
    if (selectedCount >= targetIndices.size) {
        overlayLogging("已成功选中${selectedCount}张图片（第2张和第3张）")
        return true
    }
    
    return false
}

/**
 * 尝试选中微博图片：在FrameLayout中查找并点击TextView（checkbox）
 */
private suspend fun trySelectWeiboImage(frameLayout: AccessibilityNodeInfo): Boolean {
    // 方式1: 直接查找并点击TextView（checkbox）
    val textViews = frameLayout.findByTags("android.widget.TextView")
    textViews.forEach { textView ->
        // TextView作为checkbox使用，点击它来选中图片
        showClickEffect(textView, "点击TextView(checkbox)")
        delay(100)
        
        // 尝试直接点击TextView
        if (textView.isClickable && textView.click()) {
            return true
        }
        
        // 如果TextView不可点击，尝试点击其父节点
        textView.findFirstParentClickable()?.let { parent ->
            if (parent.click()) {
                return true
            }
        }
    }
    
    // 方式2: 如果TextView点击失败，尝试点击FrameLayout本身
    if (frameLayout.isClickable) {
        showClickEffect(frameLayout, "点击FrameLayout容器")
        delay(100)
        if (frameLayout.click()) {
            return true
        }
    }
    
    // 方式3: 通过坐标手势点击
    val bounds = frameLayout.getBoundsInScreen()
    showClickEffect(bounds.centerX().toFloat(), bounds.centerY().toFloat(), "手势点击")
    delay(100)
    return frameLayout.nodeGestureClick()
}
```

### 方式2: 更精确的实现（推荐使用）

```kotlin
/**
 * 微博相册选择逻辑（优化版）
 * 根据你的描述：RecyclerView -> FrameLayout -> TextView(checkbox)
 */
private suspend fun selectWeiboImagesOptimized(): Boolean {
    overlayLogging("开始查找微博相册，准备选中第二张和第三张图片")
    val targetIndices = listOf(1, 2) // 第二张（索引1）和第三张（索引2）
    var selectedCount = 0
    
    // 查找RecyclerView
    AssistsCore.findByTags("androidx.recyclerview.widget.RecyclerView").forEach { recyclerView ->
        overlayLogging("找到RecyclerView，开始遍历FrameLayout")
        val frameLayouts = mutableListOf<AccessibilityNodeInfo>()
        
        // 遍历RecyclerView的所有子项，收集FrameLayout
        for (index in 0 until recyclerView.childCount) {
            val child = recyclerView.getChild(index) ?: continue
            
            // ⭐ 关键：检查是否为FrameLayout
            if (TextUtils.equals("android.widget.FrameLayout", child.className)) {
                frameLayouts.add(child)
            }
        }
        
        overlayLogging("共找到${frameLayouts.size}个FrameLayout")
        
        // 选中第二张和第三张（索引1和2）
        targetIndices.forEach { targetIndex ->
            if (targetIndex < frameLayouts.size) {
                val frameLayout = frameLayouts[targetIndex]
                val displayIndex = targetIndex + 1
                
                overlayLogging("选中第${displayIndex}张图片（索引${targetIndex}）")
                
                // 在FrameLayout中查找TextView（checkbox）
                frameLayout.findByTags("android.widget.TextView").forEach { textView ->
                    // ⭐ 关键：点击TextView来选中图片
                    val bounds = textView.getBoundsInScreen()
                    showClickEffect(bounds.centerX().toFloat(), bounds.centerY().toFloat(), "选中第${displayIndex}张")
                    delay(200)
                    
                    // 尝试点击TextView
                    if (textView.isClickable) {
                        if (textView.click()) {
                            selectedCount++
                            overlayLogging("成功选中第${displayIndex}张图片")
                            return@forEach
                        }
                    }
                    
                    // TextView不可点击，尝试通过父节点点击
                    textView.findFirstParentClickable()?.let { parent ->
                        if (parent.click()) {
                            selectedCount++
                            overlayLogging("通过父节点成功选中第${displayIndex}张图片")
                            return@forEach
                        }
                    }
                    
                    // 最后尝试手势点击
                    if (textView.nodeGestureClick()) {
                        selectedCount++
                        overlayLogging("通过手势成功选中第${displayIndex}张图片")
                    }
                }
                
                delay(300) // 每次选中后稍作延迟
            }
        }
        
        // 找到就退出循环
        if (frameLayouts.isNotEmpty()) return@forEach
    }
    
    val success = selectedCount >= targetIndices.size
    overlayLogging("相册选择完成：成功选中${selectedCount}/${targetIndices.size}张图片")
    return success
}
```

## 四、执行流程

```
开始
  ↓
查找 RecyclerView 根节点
  ↓
遍历 RecyclerView.childCount
  ↓
检查 child.className == "android.widget.FrameLayout"
  ↓ 是
添加到 FrameLayout 列表
  ↓
继续下一个子项
  ↓
收集完成，得到所有 FrameLayout
  ↓
选择索引1和索引2的 FrameLayout（第二张和第三张）
  ↓
对每个目标 FrameLayout：
  ├─ 查找其中的 TextView
  ├─ 尝试点击 TextView（checkbox）
  ├─ 如果失败，尝试通过父节点点击
  └─ 如果仍失败，尝试手势点击
  ↓
检查是否成功选中2张图片
  ↓
完成
```

## 五、与微信朋友圈逻辑的对比

| 对比项 | 微信朋友圈 | 微博相册 |
|--------|-----------|---------|
| **容器类型** | `RelativeLayout` | `FrameLayout` |
| **Checkbox类型** | `CheckBox` 组件 | `TextView`（作为checkbox） |
| **判断方式** | 检查是否有TextView（无TextView=图片项） | 直接使用FrameLayout |
| **选中方式** | 点击 `CheckBox` 组件 | 点击 `TextView` 节点 |

## 八、注意事项

1. **索引从0开始**：第二张图片的索引是1，第三张是2
2. **延迟等待**：每次点击后添加适当延迟，等待UI响应
3. **多种点击方式**：TextView可能不可点击，需要尝试多种方式
4. **日志记录**：添加详细日志，方便调试和排查问题
5. **容错处理**：如果一种方式失败，尝试备选方案
